snp = separate(snp, date, into = c("month", "day", "year"), sep = "/")

# convert 2-digit year to 4-digit year
snp = snp %>%
  mutate(year = if_else(year >= 0 & year <= 15, paste("20", year, sep = ""),paste("19", year, sep = ""))) %>%
